@Data

    private static final long serialVersionUID = -6013320537436191451L;
    @TableId
    @ApiModelProperty(value = "地区id", required = true)
    private Long areaId;

    @ApiModelProperty(value = "地区名称", required = true)
    private String areaName;

    @ApiModelProperty(value = "地区上级id", required = true)
    private Long parentId;

    @ApiModelProperty(value = "地区层级", required = true)

    @TableField(exist = false)
    private List<Area> areas;

# 方式一：递归查询

    @GetMapping("/all")
    public List<Area> allList() {
        //全部的地区
        List<Area> list = areaService.list();

        return list.stream()
                .filter(it -> it.getParentId() == 0)//获取全部的一级父级
                .map(it -> selectSon(it, list))
                .collect(Collectors.toList());
    }

    /**
     * 获取自己的下级
     *
     * @param parent 父级
     * @param list   全部
     * @return 父级以及全部的子级
     */
    private Area selectSon(Area parent, List<Area> list) {
        for (Area area : list) {
            if (parent.getAreaId().equals(area.getParentId())) {
                if (parent.getAreas() == null) {
                    parent.setAreas(new ArrayList<>());
                }
                parent.getAreas().add(selectSon(area, list));
            }
        }
        return parent;
    }
